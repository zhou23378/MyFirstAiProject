## 三、开发规范

### 3.1 后端规范

- **包名**: `com.salon.{模块名}`
- **分层**: entity → mapper → service(impl) → controller
- **统一响应**: 使用 `Result<T>` 和 `PageResult<T>` 封装
- **分页参数**: `page`(默认1), `pageSize`(默认10)
- **JWT认证**: 所有 `/api/**` 接口都需要 Bearer token（登录接口除外）
- **请求体**: 统一使用 `@RequestBody`（JSON格式），不支持 `application/x-www-form-urlencoded`

### 3.2 前端规范

- **API层**: 统一放在 `src/api/` 下，使用 `request.js` 封装的 axios 实例
- **响应解包**: `request.js` 的响应拦截器会自动解包 `res.data.data`，API 方法直接返回业务数据
- **路由**: 在 `src/router/index.js` 中配置，需登录的路由加 `meta: { requiresAuth: true }`
- **Token存储**: 登录后 token 存 `localStorage`，key 为 `token`
- **页面组件**: 放在 `src/views/{模块名}/` 下

### 3.3 前后端交互数据格式

**请求格式（POST/PUT）**:

```json
// Content-Type: application/json
{ "field1": "value1", "field2": "value2" }
```

**响应格式**:

```json
// 成功
{ "code": 200, "msg": "操作成功", "data": { ... } }

// 分页
{ "code": 200, "msg": "操作成功", "data": { "list": [...], "total": 100, "page": 1, "pageSize": 10, "pages": 10 } }

// 失败
{ "code": 400, "msg": "错误信息", "data": null }
```

```java
import com.salon.common.result.Result;
import com.salon.{module}.entity.{EntityName};
import com.salon.{module}.service.{EntityName}Service;

import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/{module}")
@RequiredArgsConstructor
public class {EntityName}Controller {

    private final {EntityName}Service {entityName}Service;

    @GetMapping("/page")
    @Operation(summary = "分页查询")
    public Result<PageResult<{EntityName}>> page(
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "10") int pageSize,
            @RequestParam(required = false) String keyword) {
        Page<{EntityName}> pageResult = {entityName}Service.page(
                new Page<>(page, pageSize),
                new LambdaQueryWrapper<{EntityName}>()
                        .like(keyword != null, {EntityName}::getName, keyword)
                        .orderByDesc({EntityName}::getCreateTime));
        return Result.success(PageResult.of(pageResult));
    }

    @GetMapping("/{id}")
    @Operation(summary = "根据ID查询")
    public Result<{EntityName}> getById(@PathVariable Long id) {
        return Result.success({entityName}Service.getById(id));
    }

    @PostMapping
    @Operation(summary = "新增")
    public Result<Void> create(@RequestBody {EntityName} entity) {
        {entityName}Service.save(entity);
        return Result.success();
    }

    @PutMapping("/{id}")

    public Result<Void> update(@PathVariable Long id, @RequestBody {EntityName} entity) {
        entity.setId(id);
```

### 5.2 页面组件模板

```vue
<template>
  <div class="page-container">
    <!-- 搜索栏 -->
    <el-card class="search-card" shadow="never">
      <el-form :inline="true" :model="query">
        <el-form-item label="关键词">
          <el-input v-model="query.keyword" placeholder="搜索..." clearable />
        </el-form-item>
        <el-form-item>
          <el-button type="primary" @click="fetchData">查询</el-button>
          <el-button @click="resetQuery">重置</el-button>
        </el-form-item>
      </el-form>
    </el-card>

    <!-- 操作栏 -->
    <div class="action-bar">
      <el-button type="primary" @click="openDialog()">新增</el-button>
    </div>

    <!-- 表格 -->
    <el-table :data="list" v-loading="loading" stripe>
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="name" label="名称" />
      <el-table-column label="操作" width="200" fixed="right">
        <template #default="{ row }">
          <el-button type="primary" link @click="openDialog(row)"
            >编辑</el-button
          >
          <el-button type="danger" link @click="handleDelete(row.id)"
            >删除</el-button
          >
        </template>
      </el-table-column>
    </el-table>

    <!-- 分页 -->
    <div class="pagination">
      <el-pagination
        v-model:current-page="query.page"
        v-model:page-size="query.pageSize"
        :total="total"
        layout="total, sizes, prev, pager, next"
        @current-change="fetchData"
        @size-change="fetchData"
      />
    </div>

    <!-- 新增/编辑弹窗 -->
    <el-dialog
      v-model="dialogVisible"
      :title="isEdit ? '编辑' : '新增'"
      width="500px"
    >
      <el-form :model="form" label-width="80px">
        <el-form-item label="名称">
          <el-input v-model="form.name" />
        </el-form-item>
      </el-form>
      <template #footer>
        <el-button @click="dialogVisible = false">取消</el-button>
        <el-button type="primary" @click="handleSave">保存</el-button>
      </template>
    </el-dialog>
  </div>
</template>

<script setup>
import { ref, reactive, onMounted } from "vue";
import { ElMessage, ElMessageBox } from "element-plus";
import { moduleApi } from "../../api/{module}";

const list = ref([]);
const total = ref(0);
const loading = ref(false);
const dialogVisible = ref(false);
const isEdit = ref(false);
const currentId = ref(null);

const query = reactive({ page: 1, pageSize: 10, keyword: "" });
const form = reactive({ name: "" });

async function fetchData() {
  loading.value = true;
  try {
    const res = await moduleApi.page(query);
    list.value = res.list || [];
    total.value = res.total || 0;
  } finally {
    loading.value = false;
  }
}

function resetQuery() {
  query.keyword = "";
  query.page = 1;
  fetchData();
}

function openDialog(row) {
  if (row) {
    isEdit.value = true;
    currentId.value = row.id;
    Object.assign(form, row);
  } else {
    isEdit.value = false;
    currentId.value = null;
    form.name = "";
  }
  dialogVisible.value = true;
}

async function handleSave() {
  try {
    if (isEdit.value) {
      await moduleApi.update(currentId.value, form);
      ElMessage.success("修改成功");
    } else {
      await moduleApi.create(form);
      ElMessage.success("新增成功");
    }
    dialogVisible.value = false;
    fetchData();
  } catch (e) {
    // 错误已在拦截器中处理
  }
}

async function handleDelete(id) {
  try {
    await ElMessageBox.confirm("确认删除？");
    await moduleApi.remove(id);
    ElMessage.success("删除成功");
    fetchData();
  } catch {}
}

onMounted(fetchData);
</script>
```

### 5.4 ID 策略说明

> **当前阶段（单体架构）：** 使用 `@TableId(type = IdType.AUTO)`，主键为自增 Long 类型（见 §6.4 BaseEntity）。
> **未来微服务迁移：** 可考虑切换为 UUID 或雪花 ID（Snowflake），避免多服务自增 ID 冲突。当前无须修改。
> **枚举值参考：** 所有状态和类型字段的定义以 `docs/05-数据库参考.md` 为准。

## 九、常见问题速查

### Q: 后端编译报错？

A: 检查 pom.xml 中 Java 版本是否与本地 JDK 匹配（当前用 JDK 21，配置 java.version=21）

### Q: 接口返回 401？

A: 检查请求头是否带了 `Authorization: Bearer {token}`

### Q: 接口返回 404？

A: 检查 URL 路径是否与 Controller 中 `@RequestMapping` + `@GetMapping` 拼接一致

### Q: 接口返回 415？

A: 检查请求头 `Content-Type` 是否为 `application/json`，不要用 `application/x-www-form-urlencoded`

### Q: 前端页面空白/报错？

A: 打开浏览器 F12 控制台，查看网络请求是否正常返回，检查路由配置是否正确

### Q: 修改后端代码后不生效？

A: 需要重新 `mvn package` 打包，然后重启后端进程（旧 jar 包不会自动更新）

### Q: 新增模块需要改哪些文件？

A: 后端：entity → mapper → service → service/impl → controller（5个文件）
前端：api/{module}.js → views/{module}/index.vue → router/index.js（3个文件）
数据库：schema.sql 加 CREATE TABLE（1个文件）
